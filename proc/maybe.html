pure function maybe(string) result(maybe_string)
   type(string_type), intent(in) :: string
   character(len=slen(string)) :: maybe_string
   if (allocated(string%raw)) then
      maybe_string = string%raw
   else
      maybe_string = ''
   end if
end function maybe
