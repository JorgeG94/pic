   elemental logical function is_alphanum(c)
      character(len=1), intent(in) :: c
          !! The character to test.
      is_alphanum = (c >= '0' .and. c <= '9') .or. (c >= 'a' .and. c <= 'z') &
                    .or. (c >= 'A' .and. c <= 'Z')
   end function is_alphanum
